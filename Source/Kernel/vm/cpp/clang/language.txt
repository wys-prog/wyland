buff:
  dq(0x00) ; Will define 8 bytes 
  ; This label will be used to store temporal data.

main:
  load $r1, 0xAAE90
  load $r2, 0xFECCD
  add $r1, $r2
  ; store magic number's value in buff
  store $r1, buff

  jmp __blockEEF ; infinite loop

  __blockEEF:
    nop
    nop
    xor $r1, $r1
    jmp main
